void print_std_info(struct Student std[], int cgpa[], int n){
    for(int i=0; i<n; i++){
        printf("Final Grade: %s (%f)", std[i].grade, cgpa[i]);
        printf("\nName: %s %s\nDepartment: %s\nRoll: %hd\n\n", std[i].first_name, std[i].last_name, std[i].department, std[i].roll);
    }
}

void bubble_sort(struct Student std[], int a[], int n){
    int count_swap = 0;
    for(int i=0; i<n-1; i++){
        for(int j=0; j<n-i-1; j++){
            if(a[j]>a[j+1]){
                int temp = a[j];
                a[j] = a[j+1];
                a[j+1] = temp;

                struct Student temp_st = std[j];
                std[j] = std[j+1];
                std[j+1] = temp_st;

                count_swap++;
            }
        }
    }
    printf("Bubble sort took %d swaps\n", count_swap);
    pritn_std_info(std, a, n);
}

int smallest(int a[], int lb, int n){
    int least = a[lb], least_indx = lb;
    for(int i=lb; i<n; i++){
        if(least>a[i]){
            least = a[i];
            least_indx = i;
        }
    }
    return least_indx;
}
void selection_sort(struct Student std[], int a[], int n){
    static int count_swap=0;
    if(lb>=n) return;
    int least_indx = smallest(a, lb, n);

    int temp = a[lb];
    a[lb] = a[least_indx];
    a[least_indx] = temp;

    struct Student temp_st = std[lb];
    std[lb] = std[least_indx];
    std[least_indx] = temp_st;

    count_swap++;
    selection_sort(a, lb+1, n);
    printf("Selection sort took %d swaps\n", count_swap);
    pritn_std_info(std, a, n);
}

void insertion_sort(struct Student std[], int a[], int n){
    int count_swap  = 0;
    for(int i=1; i<n; i++){
        for(int j=i-1; j>=0 && a[j+1]<a[j]; j--){
            int tmp = a[j];
            a[j] = a[j+1];
            a[j+1] = tmp;

            struct Student temp_st = std[j];
            std[j] = std[j+1];
            std[j+1] = temp_st;

            count_swap++;
        }
    }
    printf("Insertion sort took %d swaps\n", count_swap);
    pritn_std_info(std, a, n);
}

void binary_search(int a[], int n, int d){
    int count = 0, m, l=0, h = n, d;
    while(l<=h){
        m=(l+h)/2;
        count++;
        if(a[m]==d){
            printf("index: %d" , m);
            break;
        }
        else if (a[m]<d) l=m+1;
        else h=m-1;
    }
    printf("Binary search took %d searches\n", count);
    printf("Final Grade: %s (%f)", std[m].grade, cgpa[m]);
    printf("\nName: %s %s\nDepartment: %s\nRoll: %hd\n\n", std[m].first_name, std[m].last_name, std[m].department, std[m].roll);

}

void linear_search(int a[], int n, int d){
    int count = 0;
    for(--n; n>=0; n--){
        count++;
        if(a[n] == d){

}
