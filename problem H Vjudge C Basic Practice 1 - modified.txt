#include <stdio.h>

int main() {
int a,b,i,indx;
char *num[]={"one", "two","three","four","five", "six","seven", "eight","nine", "even","odd"};
scanf("%d%d", &a,&b);
for (i=a; i<=b; i++)
{
    indx=i;
    if(i>9)
    {
        indx = 9+(indx%2)+1;
    }
    printf("%s\n", num[indx-1]);
}

    return 0;
}
