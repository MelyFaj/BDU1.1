#include <stdio.h>
int main() {
    char message[101];
    short int key;
    printf("Enter non-spaced-msg<ENTER>integer key\n---------------------------------------\n");
    scanf(" %[^\n]s", &message);
    scanf("%hd", &key);
    
    for(int i=0; message[i]!='\0'; i++){
        //if u also wish to encrypt the space then comment this 'if' out
        if(message[i]==' ' && message[i+1]!='\0') i++;
        message[i] = ((((message[i]%97)+key)%26)+97-32);
    }
    printf("%s", message);

    return 0;
}
