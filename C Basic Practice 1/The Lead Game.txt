#include <stdio.h>
#include <math.h>

int main() {
  int N,W,L=0,s,t, sum_s=0, sum_t=0;
  scanf("%d",&N);

  for(int i=0; i<N; i++){
      scanf("%d%d",&s,&t);
      sum_s+=s;
      sum_t+=t;
      if(sum_s>sum_t && abs(sum_s-sum_t)>L){
        L=abs(sum_s-sum_t);
        W=1;
     }
     else if(sum_s<sum_t && abs(sum_s-sum_t)>L){
        L=abs(sum_s-sum_t);
        W=2;
     }
      
  }
    printf("%d %d",W,L);
    return 0;
}
