#include <stdio.h>
#include <string.h>

int main() {
    char username[101]; 
    
 scanf("%s", username);
 int count_uniq_char = 0;

    for (int i = 0; i < strlen(username); i++){
         int flag = 0;
         for (int j = 0; j < i; j++){
              if (username[j] == username[i])
              {
                  flag=1;
                  break;
              }
         }

         if (flag==0){
             count_uniq_char++;
         }
    }
 
   count_uniq_char%2==1? printf("IGNORE HIM!"):printf("CHAT WITH HER!");

}
