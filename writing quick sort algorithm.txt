#include<stdio.h>
//#include<time.h>
void quick_sort(int *a, int start, int end){
    if (start >= end)
    return;
//1 9 3 7
//i=1  j=3
int i, j;
    for(i=start, j=end; i<j; i++){
        if(a[i]>a[start]){
        for(; j>=0; j--){
            if(a[j]>a[start]){
                int temp = a[i];
                a[i] = a[j];
                a[j] = temp;
                break;
            }
            }
        }
    }
    int temp = a[start];
    a[start] = a[i];
    a[i] = temp;
    
    quick_sort(a, start, i-1);
    quick_sort(a, i+1, end);
}

int main(){
    int a[4] = {1,9,3,7};
    quick_sort(a, 0, 3);
    for(int i=0; i<4; i++)
    printf("%d ", a[i]);
}
