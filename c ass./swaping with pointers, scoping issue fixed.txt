#include <stdio.h>

void swap_by_address();

int main() {
    int a,b;
scanf("%d%d", &a, &b);
swap_by_address(&a,&b);
printf("%d %d\n", a, b);

    return 0;
}

void swap_by_address(int *address_of_a, int *address_of_b)
{
    int temp;
   temp = *address_of_a;
    *address_of_a = *address_of_b;
    *address_of_b = temp;
}
