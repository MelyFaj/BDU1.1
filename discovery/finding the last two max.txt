#include <stdio.h>

int main() {
     int n;
    scanf("%d", &n);
    long int a[n],x=0,y=0;

    for(int i=0; i<n; i++){
        scanf("%ld",&a[i]);
        if(x<a[i])
            x=a[i];
        else if(y<a[i])
            y=a[i];
        if(i==n-1 && y==0)
            y=a[i-1];
    }

  printf("%d %d\n", x,y);

     return 0;
}
