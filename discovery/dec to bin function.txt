    char decimal[101];
    scanf("%s", decimal);
    int binary = decimal_to_binary(decimal);
    printf("%d", binary);

    return 0;
}

int decimal_to_binary(char *decimal){
  
  int sum=0;
  int count=0;
  
  for(int i=0; ;i++){
       if(decimal[i]=='\0') break;
       count++; 
   }

int n=count;

   for(int i=0; i<n;i++){
     
       count--;
     
       if(decimal[i]=='0') 
       sum = sum + 0;
       
       else if(decimal[i]=='1') 
       sum = sum + pow(2,count);
       
        printf("%d\n", sum);
   }
   return sum;
}
