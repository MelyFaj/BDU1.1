#include <stdio.h>

int main() {
long int n;
scanf("%ld", &n);

long int width[n], w_indx[n], w_indx_printing_order[2*n], row[n][2];
long int p_n=(2*n)+1;
char passanger[p_n];

for (long int i=0, k=0, l=0; i<n; i++){
    scanf("%ld", &width[i]);
    
   
    row[k][0]=row[k][1]='x';
    k++;
    
    w_indx[i]=l+1;
    l++;
    
    for(long int j=i; j>0; j--){
    if(width[j]<width[j-1]){
        long int temp;
        temp = width[j];
        width[j] = width[j-1];
        width[j-1] = temp;
        
        temp = w_indx[j];
        w_indx[j] = w_indx[j-1];
        w_indx[j-1] = temp;
    }
    }
}
/*
for(int i=0; i<p_n-1; i++){
    printf("%c%c", row[i][0], row[i][1]);
}
*/
scanf("%s", passanger);

for(long int i=0; i<p_n-1; i++){
    if(i==0) {
        row[i][0]='0';
        w_indx_printing_order[i] = w_indx[i];
    }
        
    else if(passanger[i]=='0'){ 
        
        for(long int j=i; j<p_n-1; j++){
        if(row[i][0]=='x' && row[i][1]=='x'){
                row[i][0]='0'; 
                w_indx_printing_order[i] = w_indx[i];
                break;
        }
        } 
    }
    else if (passanger[i]=='1'){
        for(long int j=p_n-1; j>=0; j++){
        if(row[i][0]=='0' && row[i][1]=='x'){
                row[i][1]='1'; 
                w_indx_printing_order[i] = w_indx[i];
                break;
        }
        }
    }
}

for(long int i=0; i<p_n-1; i++)
    printf("%ld ", w_indx_printing_order[i]);
    return 0;
}
