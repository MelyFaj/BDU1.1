#include <stdio.h>

int main() {
long int n;
scanf("%ld", &n);

long int width[n], w_indx[n], w_indx_printing_order[n];
long int p_n=(2*n)+1;
char passanger[p_n], row[p_n];

for (long int i=0, k=0, l=0; i<n; i++){
    scanf("%ld", &width[i]);
    
   
    row[k]=row[k+1]=row[k+2]='x';
    k+=2;
    
    w_indx[i]=l+1;
    l++;
    
    for(long int j=i; j>0; j--){
    if(width[j]<width[j-1]){
        long int temp;
        temp = width[j];
        width[j] = width[j-1];
        width[j-1] = temp;
        
        temp = w_indx[j];
        w_indx[j] = w_indx[j-1];
        w_indx[j-1] = temp;
    }
    }
}

scanf("%s", passanger);

for(long int i=0; i<p_n-1; i++){
    if(i==0) {
        row[i]='0';
        w_indx_printing_order[i] = w_indx[i];
    }
        
    else if(passanger[i]=='0'){ 
        
        for(long int j=i; j<p_n-1; j++){
        if(i%2==0 && row[i]=='x' && row[i+1]=='x'){
                row[i]='0'; 
                w_indx_printing_order[i] = w_indx[i];
                break;
        }
        } 
    }
    else if (passanger[i]=='1'){
        for(long int j=p_n-1; j>=i; j++){
        if(i%2==1 && row[i]=='0' && row[i+1]=='x'){
                row[i]='1'; 
                w_indx_printing_order[i] = w_indx[i];
                break;
        }
        }
    }
}

for(long int i=0; i<n; i++)
    printf("%ld", w_indx_printing_order[i]);
    return 0;
}
